{{- $person := . -}} {{- $subtitle := "" -}} {{- $positions := default (slice) $person.Params.positions -}} {{- $current := partial "person-current-position.html" $person -}} {{- if $current -}} {{- $subtitle = partial "person-position-title.html" (dict "position" $current "is_current" true) -}} {{- else if gt (len $positions) 0 -}} {{- $latest := index (sort $positions "end_date" "desc") 0 -}} {{- $subtitle = partial "person-position-title.html" (dict "position" $latest "is_current" false) -}} {{- $kind := lower (printf "%v" (index $latest "kind")) -}} {{- if and (in (slice "phd" "mphil") $kind) (index $latest "end_date") -}} {{- $endDate := printf "%v" (index $latest "end_date") -}} {{- $yearMatches := findRE "^[0-9]{4}" $endDate 1 -}} {{- if gt (len $yearMatches) 0 -}} {{- $subtitle = printf "%s %s" $subtitle (index $yearMatches 0) -}} {{- end -}} {{- end -}} {{- end -}} {{- return $subtitle -}}
