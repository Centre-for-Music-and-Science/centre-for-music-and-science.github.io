{{- $position := . -}} {{- $startDateValue := index $position "start_date" -}} {{- $endDateValue := index $position "end_date" -}} {{- $startDate := "" -}} {{- $endDate := "" -}} {{- $range := "" -}} {{- if $startDateValue -}} {{- $startDate = printf "%v" $startDateValue -}} {{- end -}} {{- if $endDateValue -}} {{- $endDate = printf "%v" $endDateValue -}} {{- end -}} {{- if and (not $startDate) (not $endDate) -}} {{- $range = "" -}} {{- else if and $startDate (not $endDate) -}} {{- $range = printf "%s–" $startDate -}} {{- else if and (not $startDate) $endDate -}} {{- $range = $endDate -}} {{- else if eq $startDate $endDate -}} {{- $range = $startDate -}} {{- else -}} {{- $displayEnd := $endDate -}} {{- $startYearMatch := findRE "^[0-9]{4}$" $startDate 1 -}} {{- $endYearMatch := findRE "^[0-9]{4}$" $endDate 1 -}} {{- if and (gt (len $startYearMatch) 0) (gt (len $endYearMatch) 0) -}} {{- $startCentury := index (findRE "^[0-9]{2}" $startDate 1) 0 -}} {{- $endCentury := index (findRE "^[0-9]{2}" $endDate 1) 0 -}} {{- if eq $startCentury $endCentury -}} {{- $displayEnd = index (findRE "[0-9]{2}$" $endDate 1) 0 -}} {{- end -}} {{- end -}} {{- $range = printf "%s–%s" $startDate $displayEnd -}} {{- end -}} {{- return $range -}}
